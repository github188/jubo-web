快速体验

~ $ git clone https://github.com/jubolin/jubo.git

~ $ cd jubo

~ $ ./scripts/generat-kit.sh

~ $ jubo create my-openwrt-app

~ $ cd my-openwrt-app

~/my-openwrt-app $ jubo

# => Now browse to http://localhost:3000
